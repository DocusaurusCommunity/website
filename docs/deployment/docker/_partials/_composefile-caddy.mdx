name: "docusaurus"
services:
    caddy:
        build:
            context: .

        ports:

            - DOCUSAURUS_DOMAIN=docusaurus.example
            - DOCUSAURUS_EMAIL=letsencrypt@docusaurus.example

            - caddy_data:/data
            - caddy_config:/config
